class Outer
{
	private static int num=45;
	
	static class Inner //内部类
	{
		void show()
		{
			System.out.println("show run..."+num);
		}
		
		static void function()
		{
			System.out.println("function run..."+num);
		}
		
	}
	
	public void method()
	{
		Inner in=new Inner();
		in.show();
	}
}
