class WhileTest2
{
	public static void main(String[] args)
	{
		
		/*
		要求：算出1~100之间的6的倍数出现的次数。
		
		思路：
		1.每次参与除法的数值不确定
		2.每次出现的余数的值也不确定
		3.对每次除法运算的余数是否=0进行判断
		4.参加除法运算的数值有递增规律
		5.每一次都是除法运算在重复执行

		步骤:
		1.定义一个变量，记录参与除法的数据
		2.定义一个变量，记录每次出现余数=0的次数
		3.对每次除法运算的余数进行判断，使用if判断语句
		4.对参与除法运算的数据进行自增
		5.因为除法运算需要重复，所以使用while循环结构
		*/
		
		int a=1; //记录参与除法的数据
		int count=0;//记录每次出现余数=0的次数
		while(a<=100)
		{
			if(a%6==0)
			
			count++;
			a++;
		}
		System.out.println("count="+count);
		
	}
}
