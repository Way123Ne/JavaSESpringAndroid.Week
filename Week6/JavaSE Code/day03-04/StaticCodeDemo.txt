/*
静态代码块
随着类的加载而执行。而且只执行一次

作用：
     用于给类进行初始化
*/

class StaticCode
{
//  static int num=12;

  static
  {
	  int num=12;
	  num*=3;
     System.out.println("haha");
  }

//   StaticCode(){}  不是所有的类都是通过构造函数初始化的

   static void show()
   {
       System.out.println("show run");
   }
}

class StaticCodeDemo
{
	static 
	{
		System.out.println("a");
	}
   public static void main(String[] args)
   {
   //  new StaticCode().show();
   //  new StaticCode().show();
		StaticCode.show();
		System.out.println("b");
   }
}
